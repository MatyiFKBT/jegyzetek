# LNKO

## Function
create or replace FUNCTION lnko(p1 integer, p2 integer) RETURN number IS i int;
begin
    for i in reverse 1..p2 loop
        if MOD(p1,i) = 0 and MOD(p2,i)=0 then
            return i;
        end if;
    end loop;
end;

# FAKTOR
## Function
cREATE OR REPLACE FUNCTION faktor(n integer) RETURN integer IS
    i number;
    res integer := 1;
begin
    for i in 1..n loop
        res := i*res;
    end loop;
    
    return res;
end;
